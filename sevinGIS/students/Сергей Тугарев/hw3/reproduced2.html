# Блок с основными точками
base_point <- track_points[1, ] # Это база (раздевалка)
entry_point <- track_points[100, ] # Это точка входа на круг

# Расчёты
track_analyzed <- track_points %>%     
  mutate(
    dist_to_base = as.numeric(st_distance(geometry, base_point)), # расстояние по прямой от текущей точки до базы
    dist_to_entry = as.numeric(st_distance(geometry, entry_point)), # расстояние до точки входа на трассу
    dist_step = as.numeric(st_distance(geometry, lag(geometry), by_element = TRUE)), # расстояние между текщей и предыдущей точкой
    dt = as.numeric(difftime(time, lag(time), units = "secs")), # разница во времени между текущей и предыдущей точкой
    speed = (dist_step / dt) * 3.6, # скорость
    index = row_number()) # нумерация точек

# определяем участки трассы
entry_point_idx <- which(track_analyzed$dist_to_entry < 50 & track_analyzed$index > 300 & track_analyzed$index < 600)[1]
track <- track_analyzed %>%
  mutate(Участок = case_when(
    index <= 100 ~ "Подход к трассе",
    index >= 100 & index <= entry_point_idx ~ "Круг 1",
    index >= entry_point_idx & index <= (n() - 130) ~ "Круг 2",
    index >= (n() - 130) ~ "Уход с трассы"
  )) %>%
  filter(!is.na(dist_step))
# Таблица расчётов
stats <- track %>%
  st_drop_geometry() %>%
  group_by(Участок) %>%
  summarise(
    Длина_км = sum(dist_step, na.rm = TRUE) / 1000,
    Время_сек = as.numeric(difftime(max(time), min(time), units = "secs")),
    Ср_скорость_км_ч = round((sum(dist_step, na.rm = TRUE) / Время_сек) * 3.6, 1),
    Начало = min(time),   
    Конец = max(time),
    Неравномерность = round(sd(speed, na.rm = TRUE) / mean(speed, na.rm = TRUE), 2) # если коэффициент (отношение стандартного отклонения к среднему) низкий, то шли ровно, если высокий, то скорость скакала
  ) %>%
  mutate(Длина_км = round(Длина_км, 2)) # Округляем длину
