  1. Here is R-script to get BibTex record by DOI.
require(httr)
'getDOI' <- function(doi) {   
   req <- paste0("https://dx.doi.org/",doi)
   temp <- GET(req,add_headers(Accept="application/x-bibtex"))
   if (http_error(temp))
      return(1L)
   temp <- content(temp,as="text",encoding="UTF-8")
   cat(temp)
   invisible(0L)
}
getDOI("10.5067/7Q8HCCWS4I0R")
